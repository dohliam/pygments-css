## Bash

```bash
#!/bin/bash

for i in {1..100}; do
    if (( $i % 15 == 0 )); then
        echo "FizzBuzz"
    elif (( $i % 5 == 0 )); then
        echo "Buzz"
    elif (( $i % 3 == 0 )); then
        echo "Fizz"
    else
        echo $i
    fi
done
```

## Haskell

```haskell
module Main where

fizzbuzz :: Int -> String
fizzbuzz x
    | x `mod` 15 == 0 = "FizzBuzz"
    | x `mod` 3  == 0 = "Fizz"
    | x `mod` 5  == 0 = "Buzz"
    | otherwise       = show x

main = mapM (putStrLn . fizzbuzz) [1..100]
```

## Lisp

```lisp
(defun divides-by (num divisor)
    (= (mod num divisor) 0))

(dotimes (num 100)
    (write-line
      (cond
        ((and (divides-by (+ num 1) 3) (divides-by (+ num 1) 5)) "FizzBuzz")
        ((divides-by (+ num 1) 3) "Fizz")
        ((divides-by (+ num 1) 5) "Buzz")
        (t (write-to-string (+ num 1))))))
```

## Ruby

```ruby
def fizzbuzz(number)
  divisibleBy3 = (number % 3 == 0)
  divisibleBy5 = (number % 5 == 0)

  case
    when divisibleBy3 && divisibleBy5
      puts "FizzBuzz"
    when divisibleBy3
      puts "Fizz"
    when divisibleBy5
      puts "Buzz"
    else 
      puts number
  end
end

(1..100).each {|n| fizzbuzz n}
```
